\section*{Cartesian Coordinates}

\begin{center}
\tdplotsetmaincoords{70}{110}
\begin{tikzpicture}[scale=2.5,tdplot_main_coords]

% Draw axes
\draw[thick,->] (0,0,0) -- (2,0,0) node[anchor=north east]{$x$};
\draw[thick,->] (0,0,0) -- (0,2,0) node[anchor=north west]{$y$};
\draw[thick,->] (0,0,0) -- (0,0,2) node[anchor=south]{$z$};

% Define box corners
\def\xa{0.6} \def\xb{1.2}
\def\ya{0.7} \def\yb{1.1}
\def\za{0.8} \def\zb{1.3}

% Draw back edges
\draw[blue,thick] (\xa,\ya,\za) -- (\xb,\ya,\za);
\draw[blue,thick] (\xa,\ya,\za) -- (\xa,\yb,\za);
\draw[blue,thick] (\xa,\ya,\za) -- (\xa,\ya,\zb);

% Draw top back edges
\draw[blue,thick] (\xa,\ya,\zb) -- (\xb,\ya,\zb);
\draw[blue,thick] (\xa,\ya,\zb) -- (\xa,\yb,\zb);

% Draw back top corner edges
\draw[blue,thick] (\xb,\ya,\za) -- (\xb,\ya,\zb);
\draw[blue,thick] (\xa,\yb,\za) -- (\xa,\yb,\zb);

% Fill front face (semi-transparent)
\fill[blue!20,opacity=0.6] (\xb,\ya,\za) -- (\xb,\yb,\za) -- (\xb,\yb,\zb) -- (\xb,\ya,\zb) -- cycle;

% Fill right face
\fill[blue!25,opacity=0.6] (\xa,\yb,\za) -- (\xb,\yb,\za) -- (\xb,\yb,\zb) -- (\xa,\yb,\zb) -- cycle;

% Fill top face
\fill[blue!30,opacity=0.6] (\xa,\ya,\zb) -- (\xb,\ya,\zb) -- (\xb,\yb,\zb) -- (\xa,\yb,\zb) -- cycle;

% Draw front edges
\draw[blue,thick] (\xb,\ya,\za) -- (\xb,\yb,\za);
\draw[blue,thick] (\xb,\yb,\za) -- (\xa,\yb,\za);
\draw[blue,thick] (\xb,\yb,\za) -- (\xb,\yb,\zb);
\draw[blue,thick] (\xb,\ya,\zb) -- (\xb,\yb,\zb);
\draw[blue,thick] (\xa,\yb,\zb) -- (\xb,\yb,\zb);

% Point P at center
\fill[black] ({(\xa+\xb)/2},{(\ya+\yb)/2},{(\za+\zb)/2}) circle (0.8pt);
\node at ({(\xa+\xb)/2+0.1},{(\ya+\yb)/2},{(\za+\zb)/2}) {$P$};

% Dimension labels
\draw[<->,black] (\xa,\ya-0.15,\za) -- (\xb,\ya-0.15,\za);
\node at ({(\xa+\xb)/2},\ya-0.25,\za) {$dx$};

\draw[<->,black] (\xa-0.15,\ya,\za) -- (\xa-0.15,\yb,\za);
\node at (\xa-0.28,{(\ya+\yb)/2},\za) {$dy$};

\draw[<->,black] (\xb+0.12,\yb,\za) -- (\xb+0.12,\yb,\zb);
\node at (\xb+0.25,\yb,{(\za+\zb)/2}) {$dz$};

% Side labels
\node at (\xb+0.15,{(\ya+\yb)/2},{(\za+\zb)/2}) {\small Front side};
\node at ({(\xa+\xb)/2},\yb+0.15,{(\za+\zb)/2}) {\small Right side};
\node at ({(\xa+\xb)/2},{(\ya+\yb)/2},\zb+0.15) {\small Top side};

% Point coordinates
\node at (1.8,0.3,1.6) {$P(x_o, y_o, z_o)$};

\end{tikzpicture}
\end{center}

\section*{Cylindrical Coordinates ($\rho, \phi, z$)}

\begin{center}
\tdplotsetmaincoords{70}{110}
\begin{tikzpicture}[scale=2.5,tdplot_main_coords]

% Draw axes
\draw[thick,->] (0,0,0) -- (2,0,0) node[anchor=north east]{$x$};
\draw[thick,->] (0,0,0) -- (0,2,0) node[anchor=north west]{$y$};
\draw[thick,->] (0,0,0) -- (0,0,2) node[anchor=south]{$z$};

% Cylindrical element parameters
\def\ri{0.6}  % inner radius
\def\ro{0.85} % outer radius
\def\phia{30} % start angle
\def\phib{70} % end angle
\def\za{0.7}  % bottom z
\def\zb{1.2}  % top z

% Draw bottom arcs
\draw[blue,thick] ({\ri*cos(\phia)},{\ri*sin(\phia)},\za)
    arc (\phia:\phib:\ri);
\draw[blue,thick] ({\ro*cos(\phia)},{\ro*sin(\phia)},\za)
    arc (\phia:\phib:\ro);

% Draw top arcs
\draw[blue,thick] ({\ri*cos(\phia)},{\ri*sin(\phia)},\zb)
    arc (\phia:\phib:\ri);
\draw[blue,thick] ({\ro*cos(\phia)},{\ro*sin(\phia)},\zb)
    arc (\phia:\phib:\ro);

% Draw radial lines at bottom
\draw[blue,thick] ({\ri*cos(\phia)},{\ri*sin(\phia)},\za) --
                  ({\ro*cos(\phia)},{\ro*sin(\phia)},\za);
\draw[blue,thick] ({\ri*cos(\phib)},{\ri*sin(\phib)},\za) --
                  ({\ro*cos(\phib)},{\ro*sin(\phib)},\za);

% Draw radial lines at top
\draw[blue,thick] ({\ri*cos(\phia)},{\ri*sin(\phia)},\zb) --
                  ({\ro*cos(\phia)},{\ro*sin(\phia)},\zb);
\draw[blue,thick] ({\ri*cos(\phib)},{\ri*sin(\phib)},\zb) --
                  ({\ro*cos(\phib)},{\ro*sin(\phib)},\zb);

% Draw vertical edges
\draw[blue,thick] ({\ri*cos(\phia)},{\ri*sin(\phia)},\za) --
                  ({\ri*cos(\phia)},{\ri*sin(\phia)},\zb);
\draw[blue,thick] ({\ro*cos(\phia)},{\ro*sin(\phia)},\za) --
                  ({\ro*cos(\phia)},{\ro*sin(\phia)},\zb);
\draw[blue,thick] ({\ri*cos(\phib)},{\ri*sin(\phib)},\za) --
                  ({\ri*cos(\phib)},{\ri*sin(\phib)},\zb);
\draw[blue,thick] ({\ro*cos(\phib)},{\ro*sin(\phib)},\za) --
                  ({\ro*cos(\phib)},{\ro*sin(\phib)},\zb);

% Fill outer face
\fill[blue!30,opacity=0.5]
    ({\ro*cos(\phia)},{\ro*sin(\phia)},\za) --
    ({\ro*cos(\phib)},{\ro*sin(\phib)},\za) --
    ({\ro*cos(\phib)},{\ro*sin(\phib)},\zb) --
    ({\ro*cos(\phia)},{\ro*sin(\phia)},\zb) -- cycle;

% Point P
\def\phim{50} % middle angle
\def\rm{0.725} % middle radius
\def\zm{0.95} % middle z
\fill[black] ({\rm*cos(\phim)},{\rm*sin(\phim)},\zm) circle (0.8pt);
\node at ({\rm*cos(\phim)+0.12},{\rm*sin(\phim)},\zm) {$P$};

% Dimension labels
\draw[<->,black] ({\ri*cos(\phim)},{\ri*sin(\phim)},\za-0.15) --
                 ({\ro*cos(\phim)},{\ro*sin(\phim)},\za-0.15);
\node at ({(\ri+\ro)/2*cos(\phim)},{(\ri+\ro)/2*sin(\phim)},\za-0.25) {$d\rho$};

\draw[<->,black] ({\ri*cos(\phia)},{\ri*sin(\phia)},\za-0.1)
    arc (\phia:\phib:\ri-0.1);
\node at ({\ri*cos(50)},{\ri*sin(50)-0.2},\za) {$\rho d\phi$};

\draw[<->,black] ({\ro*cos(\phib)+0.08},{\ro*sin(\phib)+0.08},\za) --
                 ({\ro*cos(\phib)+0.08},{\ro*sin(\phib)+0.08},\zb);
\node at ({\ro*cos(\phib)+0.2},{\ro*sin(\phib)+0.15},{(\za+\zb)/2}) {$dz$};

\node at (1.6,0.3,1.6) {$P(\rho_o, \phi_o, z_o)$};

\end{tikzpicture}
\end{center}

\section*{Spherical Coordinates ($r, \theta, \phi$)}

\begin{center}
\tdplotsetmaincoords{70}{110}
\begin{tikzpicture}[scale=2.5,tdplot_main_coords]

% Draw axes
\draw[thick,->] (0,0,0) -- (2,0,0) node[anchor=north east]{$x$};
\draw[thick,->] (0,0,0) -- (0,2,0) node[anchor=north west]{$y$};
\draw[thick,->] (0,0,0) -- (0,0,2) node[anchor=south]{$z$};

% Spherical element parameters
\def\ri{0.8}   % inner radius
\def\ro{1.05}  % outer radius
\def\thetai{40} % start theta angle
\def\thetaf{60} % end theta angle
\def\phii{25}   % start phi angle
\def\phif{55}   % end phi angle

% Helper function for spherical coordinates
% Bottom arc (inner radius, theta_i)
\draw[blue,thick]
    ({\ri*sin(\thetai)*cos(\phii)},{\ri*sin(\thetai)*sin(\phii)},{\ri*cos(\thetai)}) --
    ({\ri*sin(\thetai)*cos(\phif)},{\ri*sin(\thetai)*sin(\phif)},{\ri*cos(\thetai)});

% Top arc (inner radius, theta_f)
\draw[blue,thick]
    ({\ri*sin(\thetaf)*cos(\phii)},{\ri*sin(\thetaf)*sin(\phii)},{\ri*cos(\thetaf)}) --
    ({\ri*sin(\thetaf)*cos(\phif)},{\ri*sin(\thetaf)*sin(\phif)},{\ri*cos(\thetaf)});

% Side arcs (phi constant)
\draw[blue,thick]
    ({\ri*sin(\thetai)*cos(\phii)},{\ri*sin(\thetai)*sin(\phii)},{\ri*cos(\thetai)}) --
    ({\ri*sin(\thetaf)*cos(\phii)},{\ri*sin(\thetaf)*sin(\phii)},{\ri*cos(\thetaf)});

\draw[blue,thick]
    ({\ri*sin(\thetai)*cos(\phif)},{\ri*sin(\thetai)*sin(\phif)},{\ri*cos(\thetai)}) --
    ({\ri*sin(\thetaf)*cos(\phif)},{\ri*sin(\thetaf)*sin(\phif)},{\ri*cos(\thetaf)});

% Outer surface edges
\draw[blue,thick]
    ({\ro*sin(\thetai)*cos(\phii)},{\ro*sin(\thetai)*sin(\phii)},{\ro*cos(\thetai)}) --
    ({\ro*sin(\thetai)*cos(\phif)},{\ro*sin(\thetai)*sin(\phif)},{\ro*cos(\thetai)});

\draw[blue,thick]
    ({\ro*sin(\thetaf)*cos(\phii)},{\ro*sin(\thetaf)*sin(\phii)},{\ro*cos(\thetaf)}) --
    ({\ro*sin(\thetaf)*cos(\phif)},{\ro*sin(\thetaf)*sin(\phif)},{\ro*cos(\thetaf)});

\draw[blue,thick]
    ({\ro*sin(\thetai)*cos(\phif)},{\ro*sin(\thetai)*sin(\phif)},{\ro*cos(\thetai)}) --
    ({\ro*sin(\thetaf)*cos(\phif)},{\ro*sin(\thetaf)*sin(\phif)},{\ro*cos(\thetaf)});

% Radial edges
\draw[blue,thick]
    ({\ri*sin(\thetai)*cos(\phii)},{\ri*sin(\thetai)*sin(\phii)},{\ri*cos(\thetai)}) --
    ({\ro*sin(\thetai)*cos(\phii)},{\ro*sin(\thetai)*sin(\phii)},{\ro*cos(\thetai)});

\draw[blue,thick]
    ({\ri*sin(\thetai)*cos(\phif)},{\ri*sin(\thetai)*sin(\phif)},{\ri*cos(\thetai)}) --
    ({\ro*sin(\thetai)*cos(\phif)},{\ro*sin(\thetai)*sin(\phif)},{\ro*cos(\thetai)});

\draw[blue,thick]
    ({\ri*sin(\thetaf)*cos(\phif)},{\ri*sin(\thetaf)*sin(\phif)},{\ri*cos(\thetaf)}) --
    ({\ro*sin(\thetaf)*cos(\phif)},{\ro*sin(\thetaf)*sin(\phif)},{\ro*cos(\thetaf)});

% Fill outer face
\fill[blue!30,opacity=0.5]
    ({\ro*sin(\thetai)*cos(\phif)},{\ro*sin(\thetai)*sin(\phif)},{\ro*cos(\thetai)}) --
    ({\ro*sin(\thetaf)*cos(\phif)},{\ro*sin(\thetaf)*sin(\phif)},{\ro*cos(\thetaf)}) --
    ({\ro*sin(\thetaf)*cos(\phii)},{\ro*sin(\thetaf)*sin(\phii)},{\ro*cos(\thetaf)}) --
    ({\ro*sin(\thetai)*cos(\phii)},{\ro*sin(\thetai)*sin(\phii)},{\ro*cos(\thetai)}) -- cycle;

% Point P
\def\rm{0.925}
\def\thetam{50}
\def\phim{40}
\fill[black] ({\rm*sin(\thetam)*cos(\phim)},{\rm*sin(\thetam)*sin(\phim)},{\rm*cos(\thetam)})
    circle (0.8pt);
\node at ({\rm*sin(\thetam)*cos(\phim)+0.15},{\rm*sin(\thetam)*sin(\phim)},{\rm*cos(\thetam)}) {$P$};

% Dimension labels
\draw[<->,black]
    ({\ri*sin(\thetam)*cos(\phim)},{\ri*sin(\thetam)*sin(\phim)-0.08},{\ri*cos(\thetam)}) --
    ({\ro*sin(\thetam)*cos(\phim)},{\ro*sin(\thetam)*sin(\phim)-0.08},{\ro*cos(\thetam)});
\node at ({\rm*sin(\thetam)*cos(\phim)},{\rm*sin(\thetam)*sin(\phim)-0.2},{\rm*cos(\thetam)})
    {$dr$};

\node at (0.5,0.3,1.1) {$rd\theta$};
\node at (0.9,0.15,0.55) {$r\sin\theta d\phi$};

\node at (1.6,0.3,1.6) {$P(r_o, \theta_o, \phi_o)$};

\end{tikzpicture}
\end{center}
